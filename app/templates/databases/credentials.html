Node.js (using pg)
const { Pool } = require('pg');

const pool = new Pool({
  user: '{{ primary_user.username }}',
  host: '{{ server.host }}',
  database: '{{ database.name }}',
  password: '{{ primary_user.password }}',
  port: {{ server.postgres_port }},
});

// Example query
pool.query('SELECT NOW()', (err, res) => {
  console.log(err, res);
  pool.end();
});
Python (using psycopg2)
import psycopg2

conn = psycopg2.connect(
    host="{{ server.host }}",
    database="{{ database.name }}",
    user="{{ primary_user.username }}",
    password="{{ primary_user.password }}",
    port="{{ server.postgres_port }}"
)

cur = conn.cursor()
cur.execute("SELECT NOW()")
result = cur.fetchone()
print(result)

cur.close()
conn.close()
Java (using JDBC)
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.Statement;

public class PostgresJdbcExample {
    public static void main(String[] args) {
        try {
            // Connect to the PostgreSQL database
            Connection conn = DriverManager.getConnection(
                "{{ jdbc_url }}", 
                "{{ primary_user.username }}", 
                "{{ primary_user.password }}"
            );
            
            // Create a statement
            Statement stmt = conn.createStatement();
            
            // Execute a query
            ResultSet rs = stmt.executeQuery("SELECT NOW()");
            
            // Process the result set
            if (rs.next()) {
                System.out.println("Current time: " + rs.getString(1));
            }
            
            // Close resources
            rs.close();
            stmt.close();
            conn.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
